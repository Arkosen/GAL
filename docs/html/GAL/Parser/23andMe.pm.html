NAME

GAL::Parser::23andMe - Parse 23andMe genotype files

VERSION

This document describes GAL::Parser::23andMe version 0.2.0

SYNOPSIS 

    my $parser = GAL::Parser::23andMe->new(file => 'genotypes.txt');

    while (my $feature_hash = $parser->next_feature_hash) {
	print $parser->to_23andMe($feature_hash) . "\n";
    }

DESCRIPTION

GAL::Parser::23andMe provides a parser for 23andMe files.

Constructor

New GAL::Parser::23andMe objects are created by the class method new. Arguments should be passed to the constructor as a list (or reference) of key value pairs. All attributes of the Parser object can be set in the call to new. An simple example of object creation would look like this:

    my $parser = GAL::Parser::23andMe->new(file => 'snp.txt');

The constructor recognizes the following parameters which will set the appropriate attributes:

* file => feature_file.txt

This optional parameter provides the filename for the file containing the data to be parsed. While this parameter is optional either it, or the following fh parameter must be set.

* fh => feature_file.txt

This optional parameter provides a filehandle to read data from. While this parameter is optional either it, or the following fh parameter must be set.

sub parse_record {
	my ($self, $record) = @_;



	# rsid chromosome position genotype
	# rs3094315  1 742429  AA
	# rs12562034 1 758311  GG
	# rs3934834  1 995669  CC
	# rs9442372  1 1008567 AA
	# rs3737728  1 1011278 AA
	
	my $seqid      = 'chr' . $record->{chromosome};
	my $source     = '23andMe';
	my $type       = 'SNV';
	my $start      = $record->{position};
	my $end        = $record->{position};
	my $score      = '.';
	my $strand     = '+';
	my $phase      = '.';
	my $id         = $record->{rsid};

        my $reference_seq ||= uc $self->fasta->seq($seqid, $start, $end);
	my %variant_hash  = map {$_ => 1} split '', $record->{genotype};
	my @variant_seqs = keys %variant_hash;

	return undef if @variant_seqs == 1 && $variant_seqs[0] eq $reference_seq;
	
	my $zygosity = scalar @variant_seqs > 1 ? 'heterozygous' : 'homozygous';

	my $attributes = {Variant_seq   => \@variant_seqs,
			  Reference_seq => [$reference_seq],
			  ID            => [$id],
			  Zygosity      => [$zygosity],
			 };

	my $feature_data = {feature_id => $id,
			    seqid      => $seqid,
			    source     => $source,
			    type       => $type,
			    start      => $start,
			    end        => $end,
			    score      => $score,
			    strand     => $strand,
			    phase      => $phase,
			    attributes => $attributes,
			   };

	return $feature_data;
}

sub reader {
    my $self = shift;

    if (! $self->{reader}) {
	my @field_names = qw(rsid chromosome position genotype);
	my $reader = GAL::Reader::DelimitedLine->new(field_names => \@field_names);
	$self->{reader} = $reader;
    }
    return $self->{reader};
}
